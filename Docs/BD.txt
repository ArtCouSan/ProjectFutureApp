CREATE DATABASE vestibulando;

USE vestibulando;

CREATE TABLE Usuario(
    id_pessoa INT(5) NOT NULL AUTO_INCREMENT,
    Nome VARCHAR(50) NOT NULL,
    Sexo VARCHAR(1) NOT NULL,
    CPF VARCHAR(14) NOT NUll,
    Dt_Nasc varchar(10) NOT NULL,
    Cidade varchar(50) NOT NULL,
    Estado varchar(2) NOT NUll,
    Senha VARCHAR(50) NOT NULL,
    Ativo TINYINT NOT NULL,
    PRIMARY KEY (id_pessoa),
    UNIQUE KEY id_pessoa (id_pessoa)
);

CREATE TABLE Prova(
    Id_prova INT(5) NOT NULL,
    Banca VARCHAR(10) NOT NULL,
    Dt_Prova varchar(10) NOT NULL,
    Versao VARCHAR(50) NOT NULL
);

CREATE TABLE Questoes(
	Numero_q INT(5) NOT NULL,
    Id_prova INT(5) NOT NULL,
    Enunciado VARCHAR(1000) NOT NULL,
    Tema VARCHAR(50) NOT NULL,
    Sub_Tema VARCHAR(50) NOT NULL,
    Dificuldade VARCHAR(10) NOT NULL,
    Resposta VARCHAR(10) NOT NULL,
    Dica VARCHAR(40) NOT NULL,
    Alternativa_a VARCHAR(100) NOT NULL,
    Alternativa_b VARCHAR(100) NOT NULL,
    Alternativa_c VARCHAR(100) NOT NULL,
    Alternativa_d VARCHAR(100) NOT NULL,
    Alternativa_e VARCHAR(100) NOT NULL,
    Ativo TINYINT NOT NULL
);
